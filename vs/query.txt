#include<bits/stdc++.h>
using namespace std;
using ll=long long int;
int main()
{
    ll t;
    cin>>t;
    while(t--){
        ll n,q;
        cin>>n>>q;
        ll a,c;
        int b;
        vector<ll>v1;
        vector<pair<int,ll>>v2;
        for(ll i=0;i<n;i++){cin>>a;v1.push_back(a);}
         for(ll i=0;i<q;i++){cin>>b;cin>>c;v2.push_back(make_pair(b,c));}
       
        for(ll i=0;i<q;i++)
        {
            ll sum=0;
            ll m=v2[i].first;
            ll k=v2[i].second;
            
            if(m==0){
                for(ll j=0;j<n;j++){
                   if(v1[j]%2==0){
                       v1[j]=v1[j]+k;
                    sum=sum+v1[j];
                    }
                    else {sum=sum+v1[j];}
                 }
            }
            else if(m==1){
                for(ll j=0;j<n;j++){
                   if(v1[j]%2!=0){
                       v1[j]=v1[j]+k;
                    sum=sum+v1[j];
                    }
                    else {sum=sum+v1[j];}
                 }
            }
            cout<<sum<<"\n";
            sum=0;
        }
        
    }
    return 0;
}
